    definir p como entero;
    si exp = 0 entonces
        p <- 1;
    sino si exp = 1 entonces
        p <- base;
    sino si exp % 2 = 0 entonces
        p <- power(base * base % m, exp/2, m);
    sino
        p <- base * power(base, exp - 1, m) % m;
    finsi finsi finsi
